2)If elseif Example:-


import java.util.Scanner;
class Ifelseif
{

{

per = obj.nextFloat();
if(per>=90)
{
 System.out.println("percentage is A");
 }
else if(per<=90 && per>=80)
{
System.out.println("percentage is B");
}
else if(per<=80 && per>=70)
{
System.out.println("percentage is C");
}
else if(per<=70 && per>=35)
{
System.out.println("Percentage is D");
 }
else
{

}

3) do while Example:-


import java.util.Scanner;
class While
{
public static void main(String args[])
{
int n;
Scanner obj = new Scanner(System.in);
System.out.println("Enter the Number:");
n = obj.nextInt();
do
{
 System.out.println(n);
n++;
}
while(n<=10);
}
}

To Check the given number is palindrome or not


import java.util.Scanner;
class Reverse
{
public static void main(String args[])
{
int n,r,rev=0,org;
Scanner obj = new Scanner(System.in);
System.out.println("Enter the Number:");
n = obj.nextInt();
org=n;
while(n!=0)
{
    r = n%10;
    rev = rev*10+r;
    n=n/10;
}
 System.out.println("Reverse of 
 "+org+" is "+rev);
if(org==rev)
{
System.out.println(" "+org+" 
is Palindrome Number");
}
 else
{
System.out.println(" "+org+" is not
 a Palindrome Number");
}
 }
}
